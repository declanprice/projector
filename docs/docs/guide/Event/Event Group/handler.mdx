# Handler

## Usage

```tsx title="src/send-welcome-email-event.handler.ts"
import { EventGroup, EventHandler, EventMessage } from '@declanprice/projector/event'

@EventGroup({
    batchSize: 10,
})
export class CustomerEventGroup {

    @EventHandler('CustomerRegistered')
    async onChange(change: EventMessage<Customer>) {
        // handle customer registered event.
    }
}
```
